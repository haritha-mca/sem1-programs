9. Write a shell program to find the sum of squares of first n numbers? (use while) 
#!/bin/bash
echo "Enter a number:"
read n
sum=0
counter=1
while((counter<=n));do
    	square=$((counter*counter))
    	sum=$((sum+square))
    	counter=$((counter+1))
done
echo "Sum of squares is $sum"

12. Write a shell program to find the factorial of a number? (Use function) 

#!/bin/bash
calculate_factorial() {
  local num=$1
  local factorial=1
  local i=1
  while [ $i -le $num ]; do
	factorial=$((factorial * i))
	i=$((i + 1))
  done
  echo $factorial
}
echo "Enter a number:"
read num
result=$(calculate_factorial $num)
echo "Factorial of $num is: $result"

13. Write a shell program to print the Fibonacci numbers upto N? 
generate_fibonacci()
{
  local n=$1
  local a=0
  local b=1
  local i=2
  echo -n "$a, $b"
  while [ $i -lt $n ]; do
	sum=$((a + b))
	echo -n ", $sum"
	a=$b
	b=$sum
	i=$((i + 1))
  done
  echo ""
}
echo "Enter the value of N for Fibonacci series:"
read n
generate_fibonacci $n

15. Read a Decimal number. Convert it to Binary and display the result. (Use while)
#!/bin/bash
echo "Enter a decimal number:"
read decimal
binary=""
while [ $decimal -gt 0 ]; do
  remainder=$((decimal % 2))
  binary="$remainder$binary"
  decimal=$((decimal / 2))
done
echo "Binary representation: $binary"
